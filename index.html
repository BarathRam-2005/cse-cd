#define MAX_STATES 100
#define MAX_SYMBOLS 3  // 0:a, 1:b, 2:ε
#define EPSILON 2

typedef struct {

    int a_count, b_count, eps_count;
    int is_final;
} NFAState;

NFAState nfa[MAX_STATES];
int state_count = 1;  // Start from 1 for q1
int start_state = 1;
int final_state = 1;



// Stack for Thompson construction
typedef struct {
    int start;
    int end;
} Frag;

Frag stack[100];
int stack_top = -1;

// Function prototypes
void init_nfa();

// Initialize NFA
void init_nfa() {
    for(int i = 0; i < MAX_STATES; i++) {
        nfa[i].a_count = nfa[i].b_count = nfa[i].eps_count = 0;
        nfa[i].is_final = 0;
        for(int j = 0; j < MAX_STATES; j++) {
            nfa[i].a_trans[j] = nfa[i].b_trans[j] = nfa[i].eps_trans[j] = 0;
        }
    }
    state_count = 1;
    stack_top = -1;
}

// Create a new state
int create_state() {
    return state_count++;
}

// Add a transition
void add_transition(int from, int to, char symbol) {
    if(symbol == 'a') {
        nfa[from].a_trans[nfa[from].a_count++] = to;
    }
    else if(symbol == 'b') {
        nfa[from].b_trans[nfa[from].b_count++] = to;
    }
    else if(symbol == 'e') {
        nfa[from].eps_trans[nfa[from].eps_count++] = to;
    }

// Stack operations

    if(stack_top < 99) {
        stack[++stack_top] = f;
    }
}

    if(stack_top >= 0) {

// Operator precedence
int precedence(char op) {
    if(op == '|' || op == '+') return 1;
    if(op == '.') return 2;
    if(op == '*') return 3;
    return 0;
}

// Convert infix to postfix
char* to_postfix(char* infix) {
    static char postfix[200];
    char stack[100];
    int stack_top = -1;
    int pi = 0;
    int len = strlen(infix);
    
    for(int i = 0; i < len; i++) {
        char c = infix[i];
        
        if(c == 'a' || c == 'b') {
            postfix[pi++] = c;
            
            // Check for implicit concatenation
            if(i + 1 < len) {
                char next = infix[i + 1];
                if(next == 'a' || next == 'b' || next == '(') {
                    while(stack_top >= 0 && precedence('.') <= precedence(stack[stack_top])) {            
                        postfix[pi++] = stack[stack_top--];
                    }
                    stack[++stack_top] = '.';
                }
            }
        }
        else if(c == '(') {
            // Check for implicit concatenation before '('
            if(i > 0) {
                char prev = infix[i - 1];
                if(prev == 'a' || prev == 'b' || prev == ')' || prev == '*') {
                    while(stack_top >= 0 && precedence('.') <= precedence(stack[stack_top])) {
                        postfix[pi++] = stack[stack_top--];
                    }
                    stack[++stack_top] = '.';
                }
            }
            stack[++stack_top] = '(';
        }
        else if(c == ')') {
            while(stack_top >= 0 && stack[stack_top] != '(') {
                postfix[pi++] = stack[stack_top--];
            }
            if(stack_top >= 0) stack_top--;  // Pop '('
        }
        else if(c == '*') {
            postfix[pi++] = c;
            
            // Check for implicit concatenation after '*'
            if(i + 1 < len) {
                char next = infix[i + 1];
                if(next == 'a' || next == 'b' || next == '(') {
                    while(stack_top >= 0 && precedence('.') <= precedence(stack[stack_top])) {
                        postfix[pi++] = stack[stack_top--];
                    }
                    stack[++stack_top] = '.';
                }
            }
        }
        else if(c == '|' || c == '+') {
            while(stack_top >= 0 && precedence(c) <= precedence(stack[stack_top])) {
                postfix[pi++] = stack[stack_top--];
            }
            stack[++stack_top] = c;
        }
    }
    
    // Pop remaining operators
    while(stack_top >= 0) {
        postfix[pi++] = stack[stack_top--];
    }
    postfix[pi] = '\0';
    
    return postfix;
}

// Apply operator using Thompson construction
void apply_operator(char op) {
    if(op == '*') {  // Kleene star
        Frag f = pop_frag();
        int s = create_state();
        int e = create_state();
        
        // ε from new start to fragment start and to new end
        add_transition(s, f.start, 'e');
        add_transition(s, e, 'e');
        
        // ε from fragment end to fragment start (loop) and to new end
        add_transition(f.end, f.start, 'e');
        add_transition(f.end, e, 'e');
        
        push_frag(create_frag(s, e));
    }
    else if(op == '|' || op == '+') {  // Union
        Frag f2 = pop_frag();
        Frag f1 = pop_frag();
        int s = create_state();
        int e = create_state();
        
        // ε from new start to both fragments
        add_transition(s, f1.start, 'e');
        add_transition(s, f2.start, 'e');
        
        // ε from both fragments to new end
        add_transition(f1.end, e, 'e');
        add_transition(f2.end, e, 'e');
        
        push_frag(create_frag(s, e));
    }
    else if(op == '.') {  // Concatenation
        Frag f2 = pop_frag();
        Frag f1 = pop_frag();
        
        // Connect end of first to start of second
        add_transition(f1.end, f2.start, 'e');
        
        push_frag(create_frag(f1.start, f2.end));
    }
}

// Build NFA from postfix expression
void build_nfa_from_postfix(char* postfix) {
    for(int i = 0; postfix[i] != '\0'; i++) {
        char c = postfix[i];
        
        if(c == 'a' || c == 'b') {
            int s = create_state();
            int e = create_state();
            add_transition(s, e, c);
            push_frag(create_frag(s, e));
        }
        else {
            apply_operator(c);
        }
    }
    
    Frag result = pop_frag();
    start_state = result.start;
    final_state = result.end;
}

// Compute epsilon transition matrix
void compute_epsilon_matrix(int epsilon[MAX_STATES][MAX_STATES]) {
    // Initialize with direct epsilon transitions
    for(int i = 1; i < state_count; i++) {
        for(int j = 1; j < state_count; j++) {
            epsilon[i][j] = 0;
        }
        // Every state has epsilon to itself
        epsilon[i][i] = 1;
    }
    
    // Add direct epsilon transitions
    for(int i = 1; i < state_count; i++) {
        for(int j = 0; j < nfa[i].eps_count; j++) {
            int to = nfa[i].eps_trans[j];
            epsilon[i][to] = 1;
        }
    }
    
    // Compute transitive closure using Warshall's algorithm
    for(int k = 1; k < state_count; k++) {
        for(int i = 1; i < state_count; i++) {
            for(int j = 1; j < state_count; j++) {
                if(epsilon[i][k] && epsilon[k][j]) {
                    epsilon[i][j] = 1;
                }
            }
        }
    }
}

// DFS for epsilon closure
void dfs_closure(int state, int visited[], int epsilon[MAX_STATES][MAX_STATES]) {
    visited[state] = 1;
    for(int i = 1; i < state_count; i++) {
        if(epsilon[state][i] && !visited[i]) {
            dfs_closure(i, visited, epsilon);
        }
    }
}

// Infix → Postfix
void infixToPostfix(char infix[], char postfix[]) {
    int i = 0, k = 0;
    char x;

    while (infix[i] != '\0') {
        if (isalnum(infix[i])) {
            postfix[k++] = infix[i];
        }
        else if (infix[i] == '(') {
            push(infix[i]);
        }
        else if (infix[i] == ')') {
            while ((x = pop()) != '(')
                postfix[k++] = x;
        }
        else {
            while (top != -1 && priority(stack[top]) >= priority(infix[i]))
                postfix[k++] = pop();
            push(infix[i]);
        }
        i++;
    }

    while (top != -1)
        postfix[k++] = pop();

    postfix[k] = '\0';
}

            }
        }
    }
